proc run_test(index)
	async_id0 = run_cmd_async program_path + "config.cfg"
	wait 1:00
	restart
	#stop_cmd_async async_id0

	started_num = test_config_file program_path + "output.txt", "<started>"
	ended_num = test_config_file program_path + "output.txt", "<ended>"

	if (started_num == index) and (ended_num == index) then
		write_test_case "ok" + index
	else
		write_test_case "failed" + index
	endif
endproc

	search_file config_file, tag
	search_file (config_file, tag)
	return result
endproc
